### Scoring function — replace the current one with this:

function scoreAttempt(heard, targetWord) {
  if (!heard || heard.trim() === '') return { score: 0, quality: 'no-speech' };
  
  const h = heard.toLowerCase().trim();
  
  // Check against ALL 3 language versions of the word
  const targets = [
    (targetWord.word_amharic || '').toLowerCase(),
    (targetWord.word_oromo || '').toLowerCase(),
    (targetWord.word_english || '').toLowerCase(),
    (targetWord.romanization || '').toLowerCase(),
  ].filter(Boolean);
  
  let bestScore = 0;
  
  for (const t of targets) {
    if (!t) continue;
    
    // Exact match
    if (h === t) { bestScore = Math.max(bestScore, 95); continue; }
    
    // Starts with (common for Amharic — SpeechRecognition often cuts off)
    if (h.startsWith(t.substring(0, Math.floor(t.length * 0.7)))) {
      bestScore = Math.max(bestScore, 82);
      continue;
    }
    
    // Levenshtein distance for close matches
    const dist = levenshtein(h, t);
    const maxLen = Math.max(h.length, t.length);
    if (maxLen === 0) continue;
    
    const similarity = 1 - (dist / maxLen);
    const score = Math.round(similarity * 100);
    bestScore = Math.max(bestScore, score);
  }
  
  // Classify quality
  let quality;
  if (bestScore >= 85) quality = 'perfect';
  else if (bestScore >= 70) quality = 'great';
  else if (bestScore >= 45) quality = 'good';
  else if (bestScore >= 20) quality = 'try-again';
  else quality = 'no-match';
  
  return { score: bestScore, quality };
}

// Simple Levenshtein distance function
function levenshtein(a, b) {
  const m = a.length, n = b.length;
  const dp = Array.from({ length: m + 1 }, (_, i) =>
    Array.from({ length: n + 1 }, (_, j) => i === 0 ? j : j === 0 ? i : 0)
  );
  for (let i = 1; i <= m; i++) {
    for (let j = 1; j <= n; j++) {
      dp[i][j] = a[i-1] === b[j-1]
        ? dp[i-1][j-1]
        : 1 + Math.min(dp[i-1][j], dp[i][j-1], dp[i-1][j-1]);
    }
  }
  return dp[m][n];
}

const CLAUDE_URL = 'https://api.anthropic.com/v1/messages';

export async function explainHomework(
  text: string,
  childName: string,
  gradeLevel: number,
  language: 'amharic' | 'oromo' | 'english'
): Promise<string> {
  const langLabel = language === 'amharic' ? 'Amharic (አማርኛ)' : language === 'oromo' ? 'Afaan Oromo' : 'English';
  
  const systemPrompt = `You are Yeqal, a helpful Ethiopian education assistant. 
You help Ethiopian parents understand their children's homework.
ALWAYS respond in ${langLabel}.
Keep explanations simple — pitched at Grade ${gradeLevel} level.
Maximum 3 sentences. Be warm and encouraging.
Never use complex academic language.`;

  const userPrompt = `${childName} is in Grade ${gradeLevel}.
Their homework says: "${text}"
Please explain in simple ${langLabel}: what is this homework asking, and what does the student need to do?`;

  try {
    const response = await fetch(CLAUDE_URL, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'x-api-key': process.env.EXPO_PUBLIC_CLAUDE_API_KEY || '',
        'anthropic-version': '2023-06-01',
        'anthropic-dangerous-direct-browser-access': 'true',
      },
      body: JSON.stringify({
        model: 'claude-haiku-4-5-20251001',
        max_tokens: 300,
        messages: [{ role: 'user', content: userPrompt }],
        system: systemPrompt,
      }),
    });

    if (!response.ok) throw new Error('API error');
    const data = await response.json();
    return data.content[0].text;
  } catch (error) {
    // Graceful fallback — never crash
    return language === 'amharic'
      ? 'ይቅርታ፣ አሁን ማስረዳት አልቻልኩም። ቃሉን ፍለጋ ውስጥ ይፈልጉ።'
      : language === 'oromo'
      ? 'Dhiifama, amma ibsuu hin dandeenye. Jecla barbaadi.'
      : 'Sorry, I could not explain this right now. Try searching for the words.';
  }
}
