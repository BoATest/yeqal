                     currentLanguage === 'oromo' ? 'om' : 'en-US';

Step 2: When recognition.onresult fires, get the transcript.

Step 3: Calculate real score by comparing transcript to target word:
  function scoreMatch(heard, target) {
    const h = heard.toLowerCase().trim();
    const t = target.toLowerCase().trim();
    
    // Exact match
    if (h === t) return 95 + Math.floor(Math.random() * 5);
    
    // Contains the word
    if (h.includes(t) || t.includes(h)) return 75 + Math.floor(Math.random() * 15);
    
    // Check each character in common (simple similarity)
    let matches = 0;
    const shorter = h.length < t.length ? h : t;
    const longer = h.length >= t.length ? h : t;
    for (let i = 0; i < shorter.length; i++) {
      if (longer.includes(shorter[i])) matches++;
    }
    const similarity = matches / longer.length;
    
    if (similarity > 0.7) return 60 + Math.floor(similarity * 30);
    if (similarity > 0.4) return 35 + Math.floor(similarity * 40);
    return 15 + Math.floor(Math.random() * 25);
  }
